rotation specified by axis and angle

rotateAxis and Angle

element.style.rotateAxis = [ axisX, axisY, axisZ ]; // Normalization is unnecessary
element.style.rotateAngle = radian;
console.log( element.style.rotateAxis ); // [ axisX, axisY, axisZ ]
console.log( element.style.rotateAngle ); // radian

rotateAxisX

element.style.rotateAxisX = axisX;
console.log( element.style.rotateAxisX ); // axisX

rotateAxisY

element.style.rotateAxisY = axisY;
console.log( element.style.rotateAxisY ); // axisY

rotateAxisZ

element.style.rotateAxisZ = axisZ;
console.log( element.style.rotateAxisZ ); // axisZ
