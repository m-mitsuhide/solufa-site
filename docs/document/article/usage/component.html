Component

Create

var MyComponent = {
  controller: function( args ) {
    return {
      geometry: {
        type: "Box",
        value: [ .3, .3, .3 ]
      },
      material: {
        type: "MeshPhong",
        value: {
          color: args.color,
          specular: "#fff"
        }
      }
    };
  },
  view: function( ctrl ) {
    return <mesh geo={ctrl.geometry} mtl={ctrl.material}/>;
    // JavaScript: return m( "mesh", { geo: ctrl.geometry, mtl: ctrl.material } );
  }
};

Use

MSX 

var myColor = "#f00";
m.render( S.document.body, <scene>
  <MyComponent color={myColor} />
</scene> );
JavaScript 
var myColor = "#f00";
m.render( S.document.body, m( "scene", [
  m( MyComponent, { color: myColor } )
]) );
